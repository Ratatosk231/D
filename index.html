This is a unclosable, unminimizable, unmovable window. It can only by closed by hitting custom button bellow, but the button is optional too.
$( document ).ready( function () { "use strict"; // initialize taskbar $( ".taskbar" ).taskbar(); // button for opening unclosable window $( ".open-unclosable" ) .button() .on( "click", function () { openUnclosableWindow(); }); // clicking button in demo window // will trigger this function function openUnclosableWindow() { $( ".unclosable" ).window({ // cannot be closed closable : false, maximizable: false, // cannot be minimized minimizable: false, // modal - by default covers all windows and taskbars modal : true, resizable : false, // no dragging out of sight draggable : false, buttons : [ // custom button for closing { click: function () { $( ".unclosable" ) .window( "close" ); }, // use native label textLocalized: "close" } ], // don't destroy window content // on close (default demos behaviour) close: null }); } });
